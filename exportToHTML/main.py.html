import pygame
from fighter import Fighter


pygame.init()

# create a game window
SCREEN_WIDTH = 1000
SCREEN_HEIGHT = 500

screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption("Brawler")

# set frame rate
clock = pygame.time.Clock()
FPS = 60

# define colors
RED = (255, 0, 0)
GREEN = (0, 255, 0)
WHITE = (255, 255, 255)

# define fighter variables
WARRIOR_SIZE = 100
WARRIOR_DATA = [WARRIOR_SIZE]
WIZARD_SIZE = 100
WIZARD_DATA = [WIZARD_SIZE]

# load background image
bg_image = pygame.image.load("assets/images/background/background.gif").convert_alpha()

# load spritesheets
warrior_sheet1 = pygame.image.load("assets/images/warrior/sprites/Idle.png").convert_alpha()
warrior_sheet2 = pygame.image.load("assets/images/warrior/sprites/Run.png").convert_alpha()
warrior_sheet3 = pygame.image.load("assets/images/warrior/sprites/Jump.png").convert_alpha()
warrior_sheet4 = pygame.image.load("assets/images/warrior/sprites/Fall.png").convert_alpha()
warrior_sheet5 = pygame.image.load("assets/images/warrior/sprites/attack1.png").convert_alpha()
warrior_sheet6 = pygame.image.load("assets/images/warrior/sprites/attack2.png").convert_alpha()
warrior_sheet7 = pygame.image.load("assets/images/warrior/sprites/Take Hit.png").convert_alpha()
warrior_sheet8 = pygame.image.load("assets/images/warrior/sprites/Take Hit - white silhouette.png").convert_alpha()
warrior_sheet9 = pygame.image.load("assets/images/warrior/sprites/Death.png").convert_alpha()

wizard_sheet1 = pygame.image.load("assets/images/wizard/sprites/Idle.png").convert_alpha()
wizard_sheet2 = pygame.image.load("assets/images/wizard/sprites/Run.png").convert_alpha()
wizard_sheet3 = pygame.image.load("assets/images/wizard/sprites/Jump.png").convert_alpha()
wizard_sheet4 = pygame.image.load("assets/images/wizard/sprites/Fall.png").convert_alpha()
wizard_sheet5 = pygame.image.load("assets/images/wizard/sprites/Attack1.png").convert_alpha()
wizard_sheet6 = pygame.image.load("assets/images/wizard/sprites/Attack2.png").convert_alpha()
wizard_sheet7 = pygame.image.load("assets/images/wizard/sprites/Take Hit.png").convert_alpha()
wizard_sheet8 = pygame.image.load("assets/images/wizard/sprites/Take Hit - white silhouette.png").convert_alpha()
wizard_sheet9 = pygame.image.load("assets/images/wizard/sprites/Death.png").convert_alpha()

# define the number of steps per animation
WARRIOR_ANIMATION_STEPS = [4, 8, 2, 2, 4, 4, 4, 4, 4]

WIZARD_ANIMATION_STEPS = [4, 8, 2, 2, 4, 4, 4, 4, 4]

# function for drawing background
def draw_bg():
    scaled_bg = pygame.transform.scale(bg_image, (SCREEN_WIDTH, SCREEN_HEIGHT))
    screen.blit(scaled_bg, (0, 0))

# function for drawing health bars
def draw_health_bars(health, x, y):
    ratio = health / 100
    pygame.draw.rect(screen, WHITE, (x - 5, y - 5, 410, 40))
    pygame.draw.rect(screen, RED, (x, y, 400, 30))
    pygame.draw.rect(screen, GREEN, (x, y, 400 * ratio, 30))


# create two instances of fighters
fighter_1 = Fighter(200, 310, WARRIOR_DATA, warrior_sheet1, warrior_sheet2, warrior_sheet3, warrior_sheet4, warrior_sheet5, warrior_sheet6, warrior_sheet7, warrior_sheet8, warrior_sheet9, WARRIOR_ANIMATION_STEPS)
fighter_2 = Fighter(700, 310, WIZARD_DATA, wizard_sheet1, wizard_sheet2, wizard_sheet3, wizard_sheet4, wizard_sheet5, wizard_sheet6, wizard_sheet7, wizard_sheet8, wizard_sheet9, WIZARD_ANIMATION_STEPS)

# game loop (keeps the window on the screen until ready to exit game)
run = True
while run:

    clock.tick(FPS)

    # draw background
    draw_bg()

# show player stats
    draw_health_bars(fighter_1.health, 20, 20)
    draw_health_bars(fighter_2.health, 580, 20)


    # move fighters
    fighter_1.move(SCREEN_WIDTH, SCREEN_HEIGHT, screen, fighter_2)


    # draw fighters
    fighter_1.draw(screen)
    fighter_2.draw(screen)

    # event handler
    for event in pygame.event.get():
     if event.type == pygame.QUIT:
      run = False


# update display
    pygame.display.update()

# exit pygame
pygame.quit()
